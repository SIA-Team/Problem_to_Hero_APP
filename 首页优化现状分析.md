### 实施步骤

#### 1. 修改 HomeScreen.js

```javascript
// 引入 API
import questionApi from '../services/api/questionApi';

// 替换硬编码数据
const [questionList, setQuestionList] = useState([]);

// 添加数据加载函数
const loadQuestions = async (pageNum = 1, isRefresh = false) => {
  try {
    setLoading(true);
    
    // 根据当前 Tab 调用不同接口
    let response;
    switch (activeTab) {
      case t('home.recommend'):
        response = await questionApi.getRecommendList({ 
          pageNum, 
          pageSize: 20 
        });
        break;
      case t('home.hotList'):
        response = await questionApi.getHotList({ 
          pageNum, 
          pageSize: 20 
        });
        break;
      // ... 其他 Tab
      default:
        response = await questionApi.getQuestions({ 
          page: pageNum, 
          pageSize: 20 
        });
    }
    
    if (response.code === 200 && response.data) {
      const newData = response.data.rows || response.data;
      
      if (pageNum === 1) {
        setQuestionList(newData);
      } else {
        setQuestionList(prev => [...prev, ...newData]);
      }
      
      setHasMore(newData.length === 20);
    }
  } catch (error) {
    console.error('加载问题列表失败:', error);
    // 显示错误提示
  } finally {
    setLoading(false);
  }
};

// 在 useEffect 中调用
useEffect(() => {
  if (activeTab) {
    loadQuestions(1);
  }
}, [activeTab]);
```
