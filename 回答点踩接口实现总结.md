// 点踩相关状态
const [answerDislikeLoading, setAnswerDislikeLoading] = useState({}); // 请求中状态

### 4. 核心处理函数

```javascript
const handleAnswerDislike = async (answerId) => {
  // 1. 参数验证
  if (!answerId) {
    showToast('操作失败：回答ID不存在', 'error');
    return;
  }

  // 2. 防止重复请求
  if (answerDislikeLoading[answerId]) {
    console.log('🚫 点踩请求进行中，忽略重复点击');
    return;
  }

  // 3. 清除之前的防抖定时器
  if (answerDislikeDebounce[answerId]) {
    clearTimeout(answerDislikeDebounce[answerId]);
  }

  // 4. 乐观更新UI
  const currentState = answerDisliked[answerId] || false;
  const newState = !currentState;
  setAnswerDisliked(prev => ({ ...prev, [answerId]: newState }));

  // 5. 300ms防抖延迟
  const debounceTimer = setTimeout(async () => {
    try {
      setAnswerDislikeLoading(prev => ({ ...prev, [answerId]: true }));
      
      // 根据状态调用不同接口
      const response = newState 
        ? await answerApi.dislikeAnswer(answerId)
        : await answerApi.undislikeAnswer(answerId);
      
      if (response?.code === 200) {
        showToast(newState ? '已踩' : '已取消踩', 'success');
      } else {
        // 失败时回滚UI状态
        setAnswerDisliked(prev => ({ ...prev, [answerId]: currentState }));
        showToast(response?.msg || '操作失败', 'error');
      }
    } catch (error) {
      // 网络错误时回滚UI状态
      setAnswerDisliked(prev => ({ ...prev, [answerId]: currentState }));
      showToast('网络错误，请稍后重试', 'error');
    } finally {
      setAnswerDislikeLoading(prev => ({ ...prev, [answerId]: false }));
      setAnswerDislikeDebounce(prev => {
        const newState = { ...prev };
        delete newState[answerId];
        return newState;
      });
    }
  }, 300);

  // 6. 保存防抖定时器引用
  setAnswerDislikeDebounce(prev => ({ ...prev, [answerId]: debounceTimer }));
};
```

## 🛡️ 防护机制

### 1. 防抖机制 (Debounce)
- **延迟**: 300ms
- **原理**: 多次快速点击只执行最后一次
- **实现**: `setTimeout` + `clearTimeout`
- **效果**: 减少服务器请求压力

### 2. 请求去重 (Request Deduplication)
- **检查**: `answerDislikeLoading[answerId]`
- **原理**: 请求进行中忽略新的点击
- **效果**: 防止并发重复请求

### 3. 乐观更新 (Optimistic Update)
- **策略**: 先更新UI，后发请求
- **回滚**: 请求失败时恢复原状态
- **效果**: 提升用户体验

### 4. 参数验证
- **检查**: `answerId` 是否存在
- **处理**: 无效参数直接返回
- **效果**: 避免无效请求

### 5. 错误处理
- **网络错误**: 回滚UI状态 + 错误提示
- **服务器错误**: 回滚UI状态 + 显示错误信息
- **效果**: 保证数据一致性

### 6. 资源清理
- **定时器清理**: 组件卸载时清理所有定时器
- **内存管理**: 防止内存泄漏
- **效果**: 优化应用性能
